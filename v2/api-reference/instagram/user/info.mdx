---
title: "Instagram User Info"
api: "GET https://piloterr.com/api/v2/instagram/user/info"
description: "Get Instagram user info"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="https://www.instagram.com/k.mbappe">
    Instagram user url : `https://www.instagram.com/k.mbappe` or `k.mbappe`
</ParamField>

### Response

<ResponseField name="user" type="User Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="id" type="string">
            Instagram user id
        </ResponseField>
        <ResponseField name="username" type="string">
            Instagram username
        </ResponseField>
        <ResponseField name="name" type="string">
            Instagram user name
        </ResponseField>
        <ResponseField name="gender" type="string">
            Gender of the user : `male` or `female`
        </ResponseField>
        <ResponseField name="avatar" type="string">
            Instagram user avatar
        </ResponseField>
        <ResponseField name="description" type="string">
            Instagram user description
        </ResponseField>
        <ResponseField name="website" type="string">
            Instagram user website
        </ResponseField>
        <ResponseField name="private" type="boolean">
            Is the user private
        </ResponseField>
        <ResponseField name="verified" type="boolean">
            Is the user verified
        </ResponseField>
        <ResponseField name="business_account" type="boolean">
            Is the user a business account
        </ResponseField>
        <ResponseField name="business_category_name" type="string">
            Instagram user business category name
        </ResponseField>
        <ResponseField name="category_name" type="string">
            Instagram user category name
        </ResponseField>
        <ResponseField name="followers" type="number">
            Instagram user followers count

        <ResponseField name="following" type="number">
            Instagram user following count
        </ResponseField>
    </Expandable>
</ResponseField>
