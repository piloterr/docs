---
title: "Leboncoin Search API"
api: "POST https://piloterr.com/api/v2/leboncoin/search_api"
description: "Search for Leboncoin ads (optimized)"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Body

<ParamField body="filters" type="json">
    Filters to apply to the search
</ParamField>

<ParamField body="owner_type" type="string" default="all">
    Owner type
</ParamField>

<ParamField body="limit" type="integer" default="35">
    Number of ads to return
</ParamField>

<ParamField body="limit_alu" type="integer" default="3">
    Number of ads to return
</ParamField>

<ParamField body="sort_by" type="string" default="relevance">
    Sort by
</ParamField>

<ParamField body="sort_order" type="string" default="desc">
    Sort order
</ParamField>

<ParamField body="offset" type="integer" default="0">
    Offset
</ParamField>

<ParamField body="extend" type="boolean" default="true">
    Extend
</ParamField>

<ParamField body="listing_source" type="string" default="direct-search">
    Listing source
</ParamField>


### Response

<ResponseField name="company" type="Company Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="total" type="integer">
            Total number of ads
        </ResponseField>
        <ResponseField name="total_all" type="integer">
            Total number of ads (all)
        </ResponseField>
        <ResponseField name="total_pro" type="integer">
            Total of professional ads
        </ResponseField>
        <ResponseField name="total_private" type="integer">
            Total of private ads
        </ResponseField>
        <ResponseField name="total_active" type="integer">
            Total of active ads
        </ResponseField>
        <ResponseField name="total_inactive" type="integer">
            Total of inactive ads
        </ResponseField>
        <ResponseField name="total_shippable" type="integer">
            Ads that can be shipped
        </ResponseField>
        <ResponseField name="max_pages" type="integer">
            Max number of pages
        </ResponseField>
        <ResponseField name="ads" type="list">
            List of ads
        </ResponseField>
    </Expandable>
</ResponseField>
