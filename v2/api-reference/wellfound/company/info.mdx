---
title: "Wellfound Company Info"
api: "GET https://piloterr.com/api/v2/wellfound/company/info"
description: "Get company information from Wellfound"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="https://wellfound.com/company/dataiku">
    A query to search for a company. It can be: **Wellfound URL** or **Wellfound slug**
</ParamField>

### Response

<ResponseField name="company" type="Company Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="id" type="string">
            The company id
        </ResponseField>
        <ResponseField name="name" type="string">
            The company name
        </ResponseField>
        <ResponseField name="logo" type="string">
            The company logo
        </ResponseField>
        <ResponseField name="headline" type="string">
            The company headline
        </ResponseField>
        <ResponseField name="description" type="string">
            The company description
        </ResponseField>
        <ResponseField name="website" type="string">
            The company website
        </ResponseField>
        <ResponseField name="company_size" type="string">
            The company size
        </ResponseField>
        <ResponseField name="hiring" type="boolean">
            The company hiring status
        </ResponseField>
        <ResponseField name="is_operating" type="boolean">
            The company operating status
        </ResponseField>
        <ResponseField name="is_incubator" type="boolean">
            The company incubator status
        </ResponseField>
        <ResponseField name="company_financials_highlights" type="object">
            <Expandable title="properties" defaultOpen={true}>
                <ResponseField name="funding_total" type="number">
                    The company total funding
                </ResponseField>
                <ResponseField name="num_funding_rounds" type="number">
                    The company number of funding rounds
                </ResponseField>
            </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>


<RequestExample>

```bash dataiku
curl --location --request GET 'https://piloterr.com/api/v2/wellfound/company/info?query=https://wellfound.com/company/dataiku' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
