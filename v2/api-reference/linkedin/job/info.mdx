---
title: "LinkedIn Job Info"
api: "GET https://piloterr.com/api/v2/linkedin/job/info"
description: "Get job info"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="3484616454">
    Internal Linkedin ID or post url `3484616454`, or `https://www.linkedin.com/jobs/view/3194295086`
</ParamField>

### Response

<ResponseField name="job" type="object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="company_name" type="string">
            The name of the company.
        </ResponseField>
        <ResponseField name="company_url" type="string">
            The url of the company.
        </ResponseField>
        <ResponseField name="company_logo" type="string">
            The logo of the company.
        </ResponseField>
        <ResponseField name="job_title" type="string">
            The title of the job.
        </ResponseField>
        <ResponseField name="job_id" type="integer">
            The unique identifier for the job posting.
        </ResponseField>
        <ResponseField name="job_url" type="string">
            The URL of the job posting on LinkedIn.
        </ResponseField>
        <ResponseField name="list_date" type="string">
            The date on which the job posting was listed.
        </ResponseField>
        <ResponseField name="location" type="string">
            The location of the job (e.g., city and country).
        </ResponseField>
        <ResponseField name="employment_type" type="string">
            The type of employment (e.g., full-time, part-time, contract, etc.).
        </ResponseField>
        <ResponseField name="industry" type="list">
            The industry of the job : information technology, computer software, etc.
        </ResponseField>
        <ResponseField name="job_functions" type="list">
            The function of the job : creative, marketing, etc.
        </ResponseField>
        <ResponseField name="seniority_level" type="string">
            The seniority level of the job : entry level, intern, etc.
        </ResponseField>
        <ResponseField name="total_applicants" type="integer">
            The total number of applicants for the job.
        </ResponseField>
        <ResponseField name="job_description" type="string">
            The description of the job.
        </ResponseField>
    </Expandable>
</ResponseField>

<RequestExample>

```bash tiktok
curl --location --request GET 'https://piloterr.com/api/v2/linkedin/job/info?query=3484616454' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
