---
title: "LinkedIn Job Search"
api: "GET https://piloterr.com/api/v2/linkedin/job/search"
description: "Get job search results from LinkedIn"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="33246798">
    Internal Linkedin ID or post url `33246798`, or `https://www.linkedin.com/company/33246798`
</ParamField>

<ParamField query="page" type="number" optional={true} placeholder="1">
    Page number to return. Default is `1`.
</ParamField>

### Response

<ResponseField name="jobs" type="List Jobs">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="company_name" type="string">
            The name of the company offering the job.
        </ResponseField>
        <ResponseField name="company_url" type="string">
            The URL of the LinkedIn page for the company.
        </ResponseField>
        <ResponseField name="job_title" type="string">
            The title of the job being offered.
        </ResponseField>
        <ResponseField name="job_id" type="integer">
            The unique identifier for the job posting.
        </ResponseField>
        <ResponseField name="job_url" type="string">
            The URL of the job posting on LinkedIn.
        </ResponseField>
        <ResponseField name="list_date" type="string">
            The date on which the job posting was listed.
        </ResponseField>
        <ResponseField name="location" type="string">
            The location of the job (e.g., city and country).
        </ResponseField>
    </Expandable>
</ResponseField>

<RequestExample>

```bash tiktok
curl --location --request GET 'https://piloterr.com/api/v2/linkedin/job/search?query=https://www.linkedin.com/company/33246798' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
