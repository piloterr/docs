---
title: "LinkedIn Post Info"
api: "GET https://piloterr.com/api/v2/linkedin/post/info"
description: "Get information about a LinkedIn post"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="7049394404154904577">
    Internal Linkedin ID or post url `7049394404154904577`, or `https://www.linkedin.com/feed/update/urn:li:activity:7049394404154904577`
</ParamField>

### Response

<ResponseField name="post" type="Post Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="id" type="string">
            The unique identifier (urn) of the post.
        </ResponseField>
        <ResponseField name="url" type="string">
            The URL of the post.
        </ResponseField>
        <ResponseField name="type_post" type="string">
            The type of the post : article, document, image, video...
        </ResponseField>
        <ResponseField name="text" type="string">
            The text content of the post.
        </ResponseField>
        <ResponseField name="text_lang" type="string">
            The language of the text content of the post.
        </ResponseField>
        <ResponseField name="hashtags" type="array">
            An array of hashtags used in the post.
        </ResponseField>
        <ResponseField name="links" type="array">
            An array of links included in the post.
        </ResponseField>
        <ResponseField name="image_url" type="string">
            The URL of the image included in the post.
        </ResponseField>
        <ResponseField name="emojis" type="array">
            An array of emojis used in the post.
        </ResponseField>
        <ResponseField name="mentioned_profiles" type="array">
            An array of profiles mentioned in the post.
        </ResponseField>
        <ResponseField name="article" type="object">
            An object containing information about the article associated with the post.
        </ResponseField>
        <ResponseField name="document" type="object">
            An object containing information about the document associated with the post.
        </ResponseField>
        <ResponseField name="like_count" type="number">
            The number of likes on the post.
        </ResponseField>
        <ResponseField name="comments_count" type="number">
            The number of comments on the post.
        </ResponseField>
        <ResponseField name="shares_count" type="number">
            The number of shares of the post.
        </ResponseField>
        <ResponseField name="views_count" type="number">
            The number of views of the post.
        </ResponseField>
        <ResponseField name="total_engagement" type="number">
            The total number of engagements (likes, comments, shares) on the post.
        </ResponseField>
        <ResponseField name="author" type="object">
            An object containing information about the author of the post.
        </ResponseField>
    </Expandable>
</ResponseField>

<RequestExample>

```bash post
curl --location --request GET 'https://piloterr.com/api/v2/linkedin/post/info?query=7049394404154904577' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
