---
title: "LinkedIn Company Info"
api: "GET https://piloterr.com/api/v2/linkedin/company/info"
description: "Get company information from LinkedIn"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="airbusgroup">
    Internal Linkedin ID or username of the target company like `airbusgroup`, `https://www.linkedin.com/company/airbusgroup`, `2734`, `https://www.linkedin.com/company/2734`
</ParamField>


### Response

<ResponseField name="company" type="Company Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="name" type="string">
            The name of the company
        </ResponseField>

        <ResponseField name="domain" type="string">
            The company's domain name
        </ResponseField>

        <ResponseField name="founded" type="number">
            The year the company was founded
        </ResponseField>

        <ResponseField name="staff_range" type="string">
            The range of staff count (e.g. `1-10`, `11-50`, `over-10k`)
        </ResponseField>

        <ResponseField name="staff_count" type="number">
            The number of staff members in the company
        </ResponseField>

        <ResponseField name="headquarter" type="object">
            The company's headquarter address information
            <Expandable title="properties" defaultOpen={false}>
                <ResponseField name="line1" type="string">
                    The first line of the address
                </ResponseField>
                <ResponseField name="city" type="string">
                    The city of the address
                </ResponseField>
                <ResponseField name="postal_code" type="string">
                    The postal code of the address
                </ResponseField>
                <ResponseField name="geographic_area" type="string">
                    The geographic area of the address
                </ResponseField>
                <ResponseField name="country_code" type="string">
                    The ISO 3166-1 alpha-2 country code of the address
                </ResponseField>
            </Expandable>
        </ResponseField>

        <ResponseField name="website" type="string">
            The company's website URL
        </ResponseField>

        <ResponseField name="industry" type="string">
            The company's industry sector
        </ResponseField>

        <ResponseField name="tagline" type="string">
            The company's tagline or slogan
        </ResponseField>

        <ResponseField name="description" type="string">
            The company's description or summary
        </ResponseField>

        <ResponseField name="logo_url" type="string">
            The URL of the company's logo image
        </ResponseField>

        <ResponseField name="specialities" type="array">
            The specialities or areas of expertise of the company
        </ResponseField>

        <ResponseField name="follower_count" type="number">
            The number of LinkedIn users following the company page
        </ResponseField>

        <ResponseField name="associated_hashtags" type="array">
            The hashtags associated with the company
        </ResponseField>

        <ResponseField name="showcase_companies" type="array">
            The showcase companies associated with the company
            <Expandable title="properties" defaultOpen={false}>
                <ResponseField name="company_id" type="number">
                    The LinkedIn ID of the showcase company
                </ResponseField>
                <ResponseField name="company_name" type="string">
                    The name of the showcase company
                </ResponseField>
                <ResponseField name="company_url" type="string">
                    The URL of the showcase company's LinkedIn page
                </ResponseField>
            </Expandable>
        </ResponseField>

        <ResponseField name="similar_companies" type="array">
            The showcase companies associated with the company
            <Expandable title="properties" defaultOpen={false}>
                <ResponseField name="company_id" type="number">
                    The LinkedIn ID of the showcase company
                </ResponseField>
                <ResponseField name="company_name" type="string">
                    The name of the showcase company
                </ResponseField>
                <ResponseField name="company_url" type="string">
                    The URL of the showcase company's LinkedIn page
                </ResponseField>
            </Expandable>
        </ResponseField>

        <ResponseField name="is_auto_generated" type="boolean">
            Indicates whether the company page is auto-generated by LinkedIn
        </ResponseField>

    </Expandable>
</ResponseField>


<RequestExample>

```bash airbusgroup
curl --location --request GET 'https://piloterr.com/api/v2/linkedin/company/info?query=airbusgroup' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```
