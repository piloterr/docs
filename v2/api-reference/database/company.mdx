---
title: "Company"
api: "GET https://piloterr.com/api/v2/company"
description: "Get data about a company with a domain name"
---

<Note>
    60 Million Companies ðŸš€
    Enrich a company by sending us its domain name. The domain name is a better identifier than the company name because it is unique and less prone to typos.
</Note>

<Card title="Search by social network" icon="square-share-nodes" href="/v2/api-reference/database/company_social">
    You can also search a company domain by social network.
</Card>

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="gucci.com">
    A domain name like `gucci.com`, `https://gucci.com` or `www.gucci.com`
</ParamField>


### Response

<ResponseField name="company" type="Company Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="name" type="string">
            The name of the company
        </ResponseField>

        <ResponseField name="domain" type="string">
            The company's domain name
        </ResponseField>

        <ResponseField name="domain_name" type="string">
            The name portion of the company's domain name
        </ResponseField>

        <ResponseField name="domain_tld" type="string">
            The top-level domain (TLD) of the company's domain name
        </ResponseField>

        <ResponseField name="business_type" type="string">
            The type of business the company engages in (if available)
        </ResponseField>

        <ResponseField name="monthly_visitors" type="string">
            The estimated monthly number of visitors to the company's website
        </ResponseField>

        <ResponseField name="phone_number" type="string">
            The company's phone number
        </ResponseField>

        <ResponseField name="revenue" type="string">
            The company's estimated annual revenue
        </ResponseField>

        <ResponseField name="staff_range" type="string">
            The estimated number of employees at the company
        </ResponseField>

        <ResponseField name="founded" type="integer">
            The year the company was founded
        </ResponseField>

        <ResponseField name="updated_at" type="string">
            The date and time the information was last updated
        </ResponseField>

        <ResponseField name="description" type="string">
            A description of the company
        </ResponseField>

        <ResponseField name="industries" type="array">
            An array of industries the company operates in
        </ResponseField>

        <ResponseField name="social_networks" type="object">
            An object containing URLs and IDs for the company's social media accounts: `facebook`, `instagram`, `linkedin`, `pinterest`, `twitter`, `youtube`, `linkedin_sales_navigator`, `instagram_id`, `facebook_id`, `twitter_id`, `youtube_id`, `pinterest_id`, `linkedin_id_alpha`, `linkedin_id_numeric`.
        </ResponseField>

        <ResponseField name="technologies" type="array">
            An array of technologies used by the company
        </ResponseField>

        <ResponseField name="technology_categories" type="array">
            An array of categories describing the technologies used by the company
        </ResponseField>

        <ResponseField name="location" type="object">
            An object containing location information for the company
        </ResponseField>

        <ResponseField name="location.address" type="string">
            The company's street address
        </ResponseField>

        <ResponseField name="location.city" type="string">
            The city where the company is located
        </ResponseField>

        <ResponseField name="location.lat" type="string">
            The latitude of the company's location
        </ResponseField>

        <ResponseField name="location.lng" type="string">
            The longitude of the company's location
        </ResponseField>

        <ResponseField name="location.postcode" type="string">
            The postal code of the company's location
        </ResponseField>

        <ResponseField name="location.country" type="string">
            The country where the company is located
        </ResponseField>

        <ResponseField name="location.country_code" type="string">
            The ISO 3166-1 alpha-2 country code of the country where the company is located
        </ResponseField>
    </Expandable>
</ResponseField>


<RequestExample>

```bash gucci.com
curl --location --request GET 'https://piloterr.com/api/v2/company?query=gucci.com' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

```bash pennylane.com
curl --location --request GET 'https://piloterr.com/api/v2/company?query=pennylane.com' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
