---
title: "Threads User Info"
api: "GET https://piloterr.com/api/v2/threads/user/info"
description: "Get Threads user info"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Piloterr private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="zuck">
    Threads user url or slug : `https://www.threads.net/@zuck` or `zuck`
</ParamField>

<ParamField query="user_id" type="string" optional={false} placeholder="314216">
    Threads user id : `314216`
</ParamField>

### Response

<ResponseField name="user" type="User Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="id" type="string">
            Threads user id
        </ResponseField>
        <ResponseField name="username" type="string">
            Threads user username
        </ResponseField>
        <ResponseField name="full_name" type="string">
            Threads user full name
        </ResponseField>
        <ResponseField name="follower_count" type="number">
            Threads user follower count
        </ResponseField>
        <ResponseField name="profile_pic_url" type="string">
            Threads user profile picture url
        </ResponseField>
        <ResponseField name="biography" type="string">
            Threads user biography
        </ResponseField>
        <ResponseField name="bio_links" type="array">
            <Expandable title="properties" defaultOpen={true}>
                <ResponseField name="url" type="string">
                    Threads user biography link url
                </ResponseField>
            </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>

<RequestExample>

```bash zuck user info
curl --location --request GET 'https://piloterr.com/api/v2/threads/user/info?query=zuck' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
